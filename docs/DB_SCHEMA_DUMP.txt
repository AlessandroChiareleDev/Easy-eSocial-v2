### VIEW public.v_s1210_contadores
 WITH ult AS (
         SELECT DISTINCT ON (s1210_cpf_envios.empresa_id, s1210_cpf_envios.per_apur, s1210_cpf_envios.cpf) s1210_cpf_envios.empresa_id,
            s1210_cpf_envios.per_apur,
            s1210_cpf_envios.cpf,
            s1210_cpf_envios.lote_num,
            s1210_cpf_envios.status
           FROM s1210_cpf_envios
          ORDER BY s1210_cpf_envios.empresa_id, s1210_cpf_envios.per_apur, s1210_cpf_envios.cpf, s1210_cpf_envios.enviado_em DESC
        )
 SELECT s.empresa_id,
    s.per_apur,
    s.lote_num,
    count(*) AS total,
    count(*) FILTER (WHERE u.status::text = 'ok'::text) AS ok,
    count(*) FILTER (WHERE u.status::text = 'erro'::text) AS erro,
    count(*) FILTER (WHERE u.status::text = 'enviando'::text) AS enviando,
    count(*) FILTER (WHERE u.status::text = 'na'::text) AS na,
    count(*) FILTER (WHERE u.status IS NULL OR (u.status::text <> ALL (ARRAY['ok'::character varying, 'erro'::character varying, 'enviando'::character varying, 'na'::character varying]::text[]))) AS pendente
   FROM s1210_cpf_scope s
     LEFT JOIN ult u ON u.empresa_id = s.empresa_id AND u.per_apur::text = s.per_apur::text AND u.cpf = s.cpf
  GROUP BY s.empresa_id, s.per_apur, s.lote_num;
